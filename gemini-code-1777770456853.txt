pipeline {
    agent any

    environment {
        // Update this to your Docker Hub username/repo
        DOCKER_IMAGE = "khanalb58/python-flask-app"
        // This ID must match the credential you create in Jenkins
        KUBE_CONFIG_ID = 'k8s-kubeconfig'
    }

    stages {
        stage('Checkout') {
            steps {
                checkout scm
            }
        }

        stage('Static Code Testing') {
            steps {
                sh 'pip install flake8'
                // Runs linter on main.py to check for syntax/logic errors
                sh 'flake8 main.py --count --select=E9,F63,F7,F82 --show-source --statistics'
            }
        }

        stage('Docker Build') {
            steps {
                script {
                    // Builds the image with the specific Jenkins Build ID
                    sh "docker build -t ${DOCKER_IMAGE}:${env.BUILD_ID} ."
                    sh "docker build -t ${DOCKER_IMAGE}:latest ."
                }
            }
        }

        stage('Deploy to Kubernetes') {
            steps {
                withCredentials([file(credentialsId: "${KUBE_CONFIG_ID}", variable: 'KUBECONFIG')]) {
                    // Applies the deployment config to your khanalb-sp26 cluster
                    sh 'kubectl apply -f deployment-dev.yaml --kubeconfig=$KUBECONFIG'
                    sh "kubectl set image deployment/flask-app flask-app=${DOCKER_IMAGE}:${env.BUILD_ID} --kubeconfig=$KUBECONFIG"
                }
            }
        }
    }
}
